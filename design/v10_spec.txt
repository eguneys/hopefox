root(R)
root_world(R, W)

world(W) :- root_world(_, W)
world(W) :- world_edge(_, _, W)

forced_reachable(R, W) :-
  root_world(R, W)

forced_reachable(R, C) :-
  world_edge(R, P, C)
  forced_reachable(R, P)

world_edge(R, P, C) :-
  expand_ready(R, P)
  $legal_world(P, C)

attacker_to_move(R, W) :-
  forced_reachable(R, W)
  $is_attacker(W)

defender_to_move(R, W) :-
  forced_reachable(R, W)
  $is_defender(W)
