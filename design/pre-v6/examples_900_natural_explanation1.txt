## Fact definitions underlying concrete relational logic

fact bishop_sacrifice
  attacks.from = occupies.square
  attacks.to = "capturable"
  occupies.piece = Bishop

fact skewer_queen_rook
  .from = attacks_through.from
  .to = attacks_through.to
  attacks_through.to = Queen
  attacks_through.to2 = Rook

fact setup_discovered_check
  .from // these from and to are assignments to the fact
